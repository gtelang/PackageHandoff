data C'gsl_permutation = C'gsl_permutation{
  c'gsl_permutation'size :: CSize,
  c'gsl_permutation'data :: Ptr CSize
} deriving (Eq,Show)
p'gsl_permutation'size p = plusPtr p 0
p'gsl_permutation'size :: Ptr (C'gsl_permutation) -> Ptr (CSize)
p'gsl_permutation'data p = plusPtr p 8
p'gsl_permutation'data :: Ptr (C'gsl_permutation) -> Ptr (Ptr CSize)
instance Storable C'gsl_permutation where
  sizeOf _ = 16
  alignment _ = 8
  peek p = do
    v0 <- peekByteOff p 0
    v1 <- peekByteOff p 8
    return $ C'gsl_permutation v0 v1
  poke p (C'gsl_permutation v0 v1) = do
    pokeByteOff p 0 v0
    pokeByteOff p 8 v1
    return ()
