data C'gsl_ieee_float_rep = C'gsl_ieee_float_rep{
  c'gsl_ieee_float_rep'sign :: CInt,
  c'gsl_ieee_float_rep'mantissa :: [CChar],
  c'gsl_ieee_float_rep'exponent :: CInt,
  c'gsl_ieee_float_rep'type :: CInt
} deriving (Eq,Show)
p'gsl_ieee_float_rep'sign p = plusPtr p 0
p'gsl_ieee_float_rep'sign :: Ptr (C'gsl_ieee_float_rep) -> Ptr (CInt)
p'gsl_ieee_float_rep'mantissa p = plusPtr p 4
p'gsl_ieee_float_rep'mantissa :: Ptr (C'gsl_ieee_float_rep) -> Ptr (CChar)
p'gsl_ieee_float_rep'exponent p = plusPtr p 28
p'gsl_ieee_float_rep'exponent :: Ptr (C'gsl_ieee_float_rep) -> Ptr (CInt)
p'gsl_ieee_float_rep'type p = plusPtr p 32
p'gsl_ieee_float_rep'type :: Ptr (C'gsl_ieee_float_rep) -> Ptr (CInt)
instance Storable C'gsl_ieee_float_rep where
  sizeOf _ = 36
  alignment _ = 4
  peek p = do
    v0 <- peekByteOff p 0
    v1 <- let s = div 24 $ sizeOf $ (undefined :: CChar) in peekArray s (plusPtr p 4)
    v2 <- peekByteOff p 28
    v3 <- peekByteOff p 32
    return $ C'gsl_ieee_float_rep v0 v1 v2 v3
  poke p (C'gsl_ieee_float_rep v0 v1 v2 v3) = do
    pokeByteOff p 0 v0
    let s = div 24 $ sizeOf $ (undefined :: CChar)
    pokeArray (plusPtr p 4) (take s v1)
    pokeByteOff p 28 v2
    pokeByteOff p 32 v3
    return ()
