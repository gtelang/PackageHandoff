data C'gsl_dht = C'gsl_dht{
  c'gsl_dht'size :: CSize,
  c'gsl_dht'nu :: CDouble,
  c'gsl_dht'xmax :: CDouble,
  c'gsl_dht'kmax :: CDouble,
  c'gsl_dht'j :: Ptr CDouble,
  c'gsl_dht'Jjj :: Ptr CDouble,
  c'gsl_dht'J2 :: Ptr CDouble
} deriving (Eq,Show)
p'gsl_dht'size p = plusPtr p 0
p'gsl_dht'size :: Ptr (C'gsl_dht) -> Ptr (CSize)
p'gsl_dht'nu p = plusPtr p 8
p'gsl_dht'nu :: Ptr (C'gsl_dht) -> Ptr (CDouble)
p'gsl_dht'xmax p = plusPtr p 16
p'gsl_dht'xmax :: Ptr (C'gsl_dht) -> Ptr (CDouble)
p'gsl_dht'kmax p = plusPtr p 24
p'gsl_dht'kmax :: Ptr (C'gsl_dht) -> Ptr (CDouble)
p'gsl_dht'j p = plusPtr p 32
p'gsl_dht'j :: Ptr (C'gsl_dht) -> Ptr (Ptr CDouble)
p'gsl_dht'Jjj p = plusPtr p 40
p'gsl_dht'Jjj :: Ptr (C'gsl_dht) -> Ptr (Ptr CDouble)
p'gsl_dht'J2 p = plusPtr p 48
p'gsl_dht'J2 :: Ptr (C'gsl_dht) -> Ptr (Ptr CDouble)
instance Storable C'gsl_dht where
  sizeOf _ = 56
  alignment _ = 8
  peek p = do
    v0 <- peekByteOff p 0
    v1 <- peekByteOff p 8
    v2 <- peekByteOff p 16
    v3 <- peekByteOff p 24
    v4 <- peekByteOff p 32
    v5 <- peekByteOff p 40
    v6 <- peekByteOff p 48
    return $ C'gsl_dht v0 v1 v2 v3 v4 v5 v6
  poke p (C'gsl_dht v0 v1 v2 v3 v4 v5 v6) = do
    pokeByteOff p 0 v0
    pokeByteOff p 8 v1
    pokeByteOff p 16 v2
    pokeByteOff p 24 v3
    pokeByteOff p 32 v4
    pokeByteOff p 40 v5
    pokeByteOff p 48 v6
    return ()
